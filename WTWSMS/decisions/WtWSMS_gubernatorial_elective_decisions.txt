title_decisions = {
	toggle_vote_in_gub_elective_voting_for_title = {
		filter = sub_realm_owned
		ai_target_filter = self
	
		only_playable = yes
		
		from_potential = {
			ai = no
		}
		
		potential = {
			has_law = succ_gubernatorial_elective
		}
		
		effect = { #ROOT is the title
			if = {
				limit = { has_title_flag = vote_in_gub_elective_off }
				clr_title_flag = vote_in_gub_elective_off
			}
			else = { set_title_flag = vote_in_gub_elective_off }
		}
		
		ai_will_do = {
			factor = 0
		}
	}
}
